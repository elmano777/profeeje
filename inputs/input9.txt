var int limiteGlobal;
var bool ejecutar;

fun int procesar(int base):
    var int temp;
    temp = base * 2;
    return temp
endfun

fun void main():
    var int i;
    limiteGlobal = 100;
    ejecutar = true;
    i = 1;
    while i < 10 do
        i = procesar(i);
        i++
    endwhile;
    return 0
endfun
